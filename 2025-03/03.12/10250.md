### 기존 코드

```js
const fs = require('fs');
const stdin = fs.readFileSync('input.txt').toString().split('\n');

const hotelInfo = stdin.splice(1);

const hotelInfoArray = hotelInfo.map((v) => v.split(' ').map(Number));

function solution() {
  hotelInfoArray.forEach((hotelInfo) => {
    const [h, w, n] = hotelInfo;

    let floor = n % h;
    if (floor === 0) {
      floor = h;
    }
    let roomNumber = Math.floor((n - 1) / h) + 1;
    console.log(`${floor}${String(roomNumber).padStart(2, '0')}`);
  });
}

solution();
```
